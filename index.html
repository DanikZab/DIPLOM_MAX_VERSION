Приобретенные навыки

Изучил правильное использование HTML5 тегов: <header>, <nav>, <section>, <article>, <aside>, <footer>.

<article class="blog-post">
    <header>
        <h2>Заголовок статьи</h2>
        <p class="meta">Опубликовано <time datetime="2023-05-15">15 мая 2023</time></p>
    </header>
    <section class="content">
        <p>Основное содержание статьи...</p>

    </footer>
</article>

Освоил современные методы верстки с использованием CSS Grid и Flexbox для создания адаптивных макетов.

.container {
    display: grid;
    grid-template-columns: repeat(auto-fill, minmax(250px, 1fr));
    gap: 20px;
}

.card {
    display: flex;
    flex-direction: column;
    justify-content: space-between;
}

Научился создавать плавные анимации и переходы для улучшения пользовательского опыта.

.button {
    transition: all 0.3s ease;
    transform: scale(1);
}

.button:hover {
    transform: scale(1.05);
    box-shadow: 0 5px 15px rgba(0,0,0,0.1);
}

Научился работать с DOM, изменять элементы страницы динамически.

// Добавление нового элемента
const newElement = document.createElement('div');
newElement.textContent = 'Новый элемент';
newElement.classList.add('box');
document.body.appendChild(newElement);

// Обработка событий
document.querySelector('.btn').addEventListener('click', function() {
    alert('Кнопка нажата!');
});

Освоил работу с асинхронными запросами к серверу.

// Получение данных с сервера
fetch('https://api.example.com/data')

Познакомился с серверной разработкой на Node.js.

// Простой сервер на Express
const express = require('express');
const app = express();
const port = 3000;

app.get('/', (req, res) => {
    res.send('Привет, мир!');
});

app.listen(port, () => {
    console.log(`Сервер запущен на http://localhost:${port}`);
});

Изучил основы работы с MongoDB и SQL.

// Пример работы с MongoDB
const MongoClient = require('mongodb').MongoClient;
const url = 'mongodb://localhost:27017';
const dbName = 'myproject';

MongoClient.connect(url, function(err, client) {
    const db = client.db(dbName);
    const collection = db.collection('documents');
    
    collection.insertOne({name: 'Тестовый документ'}, function(err, result) {
        console.log('Документ добавлен');
        client.close();
    });
});
